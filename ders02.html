ğŸ“ Ã–rnek 2: Ownership ve ShadowingExample 2: Ownership and Shadowing

BERK (Ownership)
function main() {
    // Shadowing - same name, new binding
    let x = 5;
    let x = x + 1;      // New binding, shadows previous
    let x = x * 2;      // x is now 12
    print("Shadowed x: {}", x);
    
    // Type change via shadowing
    let spaces = "   ";
    let spaces = spaces.len();  // Now usize, not &str
    print("Space count: {}", spaces);
    
    // Move semantics
    let s1 = String::from("hello");
    let s2 = s1;  // s1 is moved, no longer valid
    // print("{}", s1);  // ERROR: value borrowed after move
    print("s2: {}", s2);
    
    // Clone for deep copy
    let s3 = String::from("world");
    let s4 = s3.clone();  // Deep copy
    print("s3: {}, s4: {}", s3, s4);  // Both valid
    
    // Copy types (stack-only, automatic copy)
    let a = 42;
    let b = a;  // Copy, not move
    print("a: {}, b: {}", a, b);  // Both valid
}
